00182 // get a UTF8 character; leave the tracking pointer at the start of the
00183 // next character
00184 // not protected against invalid UTF8
00185 template <typename T>
00186 inline void FTUnicodeStringItr<T>::readUTF8()
00187 {
00188     unsigned int ch = 0;
00189     unsigned int extraBytesToRead = utf8bytes[(unsigned char)(*nextPos)];
00190     // falls through
00191     switch (extraBytesToRead)
00192     {
00193           case 6: ch += *nextPos++; ch <<= 6; /* remember, illegal UTF-8 */
00194           case 5: ch += *nextPos++; ch <<= 6; /* remember, illegal UTF-8 */
00195           case 4: ch += *nextPos++; ch <<= 6;
00196           case 3: ch += *nextPos++; ch <<= 6;
00197           case 2: ch += *nextPos++; ch <<= 6;
00198           case 1: ch += *nextPos++;
00199     }
00200     ch -= offsetsFromUTF8[extraBytesToRead-1];
00201     curChar = ch;
00202 }

00218 template <typename T>
00219 inline void FTUnicodeStringItr<T>::readUTF16()
00220 {
00221     unsigned int ch = *nextPos++;
00222     // if we have the first half of the surrogate pair
00223     if (ch >= highSurrogateStart && ch <= highSurrogateEnd)
00224     {
00225         unsigned int ch2 = *curPos;
00226         // complete the surrogate pair
00227         if (ch2 >= lowSurrogateStart && ch2 <= lowSurrogateEnd)
00228         {
00229             ch = ((ch - highSurrogateStart) << highSurrogateShift)
00230                 + (ch2 - lowSurrogateStart) + lowSurrogateBase;
00231             ++nextPos;
00232         }
00233     }
00234     curChar = ch;
00235 }
