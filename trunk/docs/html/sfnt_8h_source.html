00228   /*                                                                       */
00229   /*                If *length != 0:                                       */
00230   /*                  Loads the next `length' bytes of table or font,      */
00231   /*                  starting at offset `offset' (in table or font too).  */
00232   /*                                                                       */
00233   /* <Output>                                                              */
00234   /*    buffer :: The address of target buffer.                            */
00235   /*                                                                       */
00236   /* <Return>                                                              */
00237   /*    TrueType error code.  0 means success.                             */
00238   /*                                                                       */
00239   typedef FT_Error

00241                        FT_ULong   tag,
